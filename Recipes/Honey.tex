\begin{Large}
    Honey
\end{Large}

\begin{scriptsize}
\begin{multicols}{2}
[
\vspace{1em}
Yields: 1 Unit
\vspace{-1.5em}
]

\begin{itemize}
    \item 256 g water
    \item 4 g flower
    \item 200 g sugar
    \item 1/4 tsp cream of tartar
\end{itemize}
\end{multicols}
\end{scriptsize}

\begin{footnotesize}
\begin{enumerate}
    \item Brew a tea with the flowers and boil water. Steep for 10 minutes
    \item Strain tea in to a small pan and bring to a boil
    \item Add sugar and stir to dissolve
    \item After 1 minute, add cream of tartar
    \item Cook for 3 minute
\end{enumerate}
\end{footnotesize}
