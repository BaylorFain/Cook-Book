\begin{Large}
    Oat Meal
\end{Large}

\begin{scriptsize}
\begin{multicols}{2}
[
\vspace{1em}
Yields: 1 Units
\vspace{-1.5em}
]

Ingredients
\begin{itemize}
    \item 26g Quick Oats
    \item 117g Water
    \item 26g Raisins
\end{itemize}
\end{multicols}
\end{scriptsize}

\begin{footnotesize}
\begin{enumerate}
    \item Combine oats and water
    \item Microwave for 1:30-2:00 mins just before it boils over
    \item Add raisins
    \item Let set for 2-5 mins, to allow the oats to thicken
    \item Please enjoy
\end{enumerate}
\end{footnotesize}
